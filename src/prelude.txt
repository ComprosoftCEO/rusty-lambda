; Boolean
true = \x y.x
false = \x y.y

not = \x.(x false true)
and = \x y.(x y false)
or  = \x y.(x true y)
if  = \p l r.(p l r)

; Numbers
succ = \n.\f x.((n f) (f x))
add  = \n m.(n succ m)
mul  = \n m.(n (add m) 0)
pow  = \n m.(m (mul n) 1)

; Pairs
pair  = \x y.\s.(s x y)
left  = \p.(p true)
right = \p.(p false)

; Lists
nil  = (pair false false)
cons = \h t.(pair true (pair h t))

empty? = \l.(not (left l))
head   = \l.(left (right l))
tail   = \l.(right (right l))
